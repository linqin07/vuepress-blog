```sql
SELECT name,
MAX(case kecheng WHEN '语文' THEN fenshu ELSE 0 END) '语文',
MAX(case kecheng WHEN '数学' THEN fenshu ELSE 0 END) '数学',
MAX(case kecheng WHEN '英语' THEN fenshu ELSE 0 END) '英语',
SUM(fenshu) '总分'
from test
GROUP BY NAME
```
